  """Sorts an array in ascending order using the bubble sort algorithm.

  Args:
    array: The array to be sorted.

  Returns:
     The sorted array.
  """

  n = len(array)

  for i in range(n):
    for j in range(0, n - i - 1):
      if array[j] > array[j + 1]:
